class UserChannel(db.Model):
    """
    用户关注频道表
    """
    __tablename__ = 'news_user_channel'

    id = db.Column('user_channel_id', db.Integer, primary_key=True, doc='主键ID')
    user_id = db.Column(db.Integer, doc='用户ID')
    channel_id = db.Column(db.Integer, doc='频道ID')
    sequence = db.Column(db.Integer, default=0, doc='序号')
    is_deleted = db.Column(db.Boolean, default=False, doc='是否删除')

from flask import request
from utils.decorators import login_required
from sqlalchemy.dialects.mysql import insert


class UserChannelResource(Resource):
    """用户频道"""
    method_decorators = {'put': [login_required]}

    def put(self):
        """修改用户频道  重置式更新"""

        UserChannel.query.filter(UserChannel.user_id == userid, UserChannel.is_deleted == False).update({'is_deleted': True})

        # 更新数据   执行的SQL多, 效率太低

        #     # 查询频道数据
        #     channel = UserChannel.query.options(UserChannel.id).filter(UserChannel.user_id == userid, UserChannel.channel_id == channel['id'])
        #

        #         # 如果频道在用户频道表中, 修改记录  sequence  is_deleted
        #         pass
        #     else:
        #         # 如果频道没有在用户频道表中, 添加记录  user_id  channel_id  is_deleted  sequence
        #         pass

        """使用mysql的优化语法  一句批量插入&更新"""

        # 批量插入
        insert_stmt = insert(UserChannel).values([{'user_id': userid, 'channel_id': channel['id'], 'sequence': channel['seq']} for channel in channels])
        # 批量更新
        update_stmt = insert_stmt.on_duplicate_key_update(is_deleted=False, sequence=insert_stmt.inserted.sequence)

        # 执行sql
        db.session.execute(update_stmt)

        # 提交事务
        db.session.commit()

        return {'channels': channels}

  • 注意点:

    • 修改用户频道需要用户登录, 注意设置类装饰器进行 权限控制
    • 重置式更新 要求先将 当前用户关注的频道进行逻辑删除

    • mysql提供了语法 ON DUPLICATE KEY UPDATE:

      • 可以实现数据 插入或更新
      • 机制为: 默认会添加数据, 如果添加时出现唯一约束的冲突, 则更新数据
      • 该语法还支持批量处理, 可以减少执行的SQL条数
      # 添加数据, 如果出现唯一约束的冲突, 则更新数据
ALTER TABLE t_user UNIQUE ('name');
INSERT INTO t_user (name, pwd, type) VALUES ('zs', 'xxx', 1) ON DUPLICATE KEY UPDATE pwd='xxxx', type=1; 

# 批量插入/更新
INSERT INTO t_user (name, pwd, type) VALUES ('zs', 'xxx', 1), ('ls', 'xxx', 1), ('ww', 'xxx', 1) ON DUPLICATE KEY UPDATE pwd=values(pwd), type=1;
