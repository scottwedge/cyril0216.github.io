# 导入: 
from rest_framework import mixins
from rest_framework.viewsets import  GenericViewSet
from . import serializers
from rest_framework.permissions import IsAuthenticated
from rest_framework.response import Response
from rest_framework import status
from rest_framework.decorators import action

# 添加 AddressViewSet 视图集
# 这个视图集我们继承自 CreateModelMixin和UpdateModelMixin
# 所以我们有了 create函数 和 update函数, 所以可以重写这两个函数
# 另外我们还是试图集,所以需要继承自一个试图集,我们这里继承自: GenericViewSet
class AddressViewSet(
    mixins.ListModelMixin,
    mixins.DestroyModelMixin,
    mixins.CreateModelMixin, 
    mixins.UpdateModelMixin, 
    GenericViewSet):
    """
    定义了如下接口:
    地址的增删改查接口&更改标题接口&设置默认地址接口
    """
    # 设置序列化器类:
    serializer_class = serializers.UserAddressSerializer
    # 设置权限: 只有校验过的用户才能够进行访问
    permissions = [IsAuthenticated]



    # 我们重写了一下 get_queryset 函数, 原因:
    # 这里前端没有给我们传递 pk 参数, 所以我们不知道获取哪个用户的地址信息
    # 故, 查询集就没有值, 但是查询集没有值会报错, 所以我们只能强制给他添加一个
    # 视图集, 但是添加哪个人的视图集呢? 哦,想起来了, 我这里进行了校验, 只有取得
    # 校验权限的人才能够访问我们的接口, 所以, 我们把 校验过的用户获取出来, 即:
    # self.request.user, 拿到之后, 根据我们之前学的定理: django默认情况下每一个主表的对象都有一个外键的属性. 如果这个外键是通过relatedj_name来定义的,则可以省略 _set 
    def get_queryset(self):
        # 我们这里通过外键的属性, 来获取有权限的用户的地址信息:
        return self.request.user.addresses.filter(is_deleted=False)



    # GET /addresses/
    def list(self, request, *args, **kwargs):
        """
        刚进入页面就获取当前用户的所有地址信息, 展示
        """
        # 获取所有的地址信息:
        queryset = self.get_queryset()
        # 将多个地址信息序列化:
        serializer = self.get_serializer(queryset, many=True)
        # 获取通过验证的用户:
        user = self.request.user

        # 拼接字典,返回给前端, 展示当前用户所有的地址信息.
        return Response({
            'user_id': user.id,
            'default_address_id': user.default_address_id,
            'limit': 20,  # 用户添加收货地址最大数量 20
            'addresses': serializer.data,
        })



    # POST /addresses/
    def create(self, request, *args, **kwargs):
        """
        保存用户地址数据
        """
        # 检查用户地址数据数目是否超过上限:
        count = request.user.addresses.count()
         # 用户添加收货地址最大数量 20
        if count >= 20:
            return Response({'message': '保存地址数据已达到上限'}, status=status.HTTP_400_BAD_REQUEST)
        # 如果不超过, 调用父类的保存方法保存
        return super().create(request, *args, **kwargs)



    # delete /addresses/<pk>/
    def destroy(self, request, *args, **kwargs):
        """
        处理删除
        """
        # 根据 pk 获取单个对象
        address = self.get_object()
        # 进行逻辑删除,修改数据库中的字段
        address.is_deleted = True
        # 修改完后保存
        address.save()
        # 返回响应
        return Response(status=status.HTTP_204_NO_CONTENT)



    # put /addresses/pk/status/
    # 这里 methods 为 put, 代表需要使用 put 方法请求当前接口
    # detail 为 True, 代表需要 pk 值, 所以路径上增加了 pk 值.
    @action(methods=['put'], detail=True)
    def status(self, request, pk=None):
        """
        设置默认地址
        """
        # 根据 pk 获取单个对象
        address = self.get_object()
        # 将获取的单个地址, 赋给 User表 中记录的默认地址字段
        request.user.default_address = address
        # 保存修改
        request.user.save()
        # 返回响应
        return Response({'message': 'OK'}, status=status.HTTP_200_OK)



    # put /addresses/pk/title/
    # 这里 methods 为 put, 代表需要使用 put 方法请求当前接口
    # detail 为 True, 代表需要 pk 值, 所以路径上增加了 pk 值.
    @action(methods=['put'], detail=True)
    def title(self, request, pk=None):
        """
        修改标题
        """
        # 根据 pk 获取单个对象
        address = self.get_object()
        # 调用自定义的用户标题序列化器:
        # 对用户传过来的标题进行反序列化, 然后更新保存到上一步获取的address中.
        serializer = serializers.AddressTitleSerializer(address, data=request.data)
        # 进行验证, 是否有格式问题:
        serializer.is_valid(raise_exception=True)
        # 验证过后保存
        serializer.save()
        # 返回响应
        return Response(serializer.data)

# 导入:
from .models import Address
import re

class UserAddressSerializer(serializers.ModelSerializer):
    """
    用户地址序列化器
    """
    province = serializers.StringRelatedField(read_only=True)
    city = serializers.StringRelatedField(read_only=True)
    district = serializers.StringRelatedField(read_only=True)
    province_id = serializers.IntegerField(label='省ID', required=True)
    city_id = serializers.IntegerField(label='市ID', required=True)
    district_id = serializers.IntegerField(label='区ID', required=True)

    class Meta:
        model = Address
        exclude = ('user', 'is_deleted', 'create_time', 'update_time')

    def validate_mobile(self, value):
        """
        验证手机号
        """
        if not re.match(r'^1[3-9]\d{9}$', value):
            raise serializers.ValidationError('手机号格式错误')
        return value

    def create(self, validated_data):
        """
        保存
        """
        validated_data['user'] = self.context['request'].user
        return super().create(validated_data)
