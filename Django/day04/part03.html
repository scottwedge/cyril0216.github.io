Introduction

使用 Django 开发 REST API 接口

我们以 Django 框架中使用的图书英雄案例来写一套支持图书数据增删改查的 REST API 接口

来理解 REST API 的开发

在此案例中，前后端均默认发送 JSON 格式数据:

在 booktest 子应用的 views.py 中:
# views.py
class BookListView(View):
    """查询所有图书、增加图书"""

    def get(self, request):
        """
        查询所有图书
        路由：GET /books/
        """

        # 从数据库中 获取所有的图书信息:
        queryset = BookInfo.objects.all()
        # 创建一个新的列表,用于存储查询出来的每一个图书数据
        book_list = []
        # 遍历查询集, 获取每一本书, 并且拼接, 存放到 list 中
        for book in queryset:
            book_list.append({
                'id': book.id,
                'btitle': book.btitle,
                'bpub_date': book.bpub_date,
                'bread': book.bread,
                'bcomment': book.bcomment,
                # 判断 image 是否存在, 如果不存在,则为 ''
                'image': book.image.url if book.image else ''
            })

        # 返回json类型数据
        # 这里 JsonResponse(dict, safe=True) 
        # 默认第一个参数接收 dict 类型
        # 如果第一个参数不是字典类型
        # 需要把 safe=False, 否则报类型错误
        return JsonResponse(book_list, safe=False)

    def post(self, request):
        """
        新增图书: 用户传入要新增的数据
        路由：POST /books/  参数类型: json
        """

        # 默认使用 json 传输,所以调用 request.body 得到 bytes
        json_bytes = request.body
        # 解码: 把 bytes 转为 str
        json_str = json_bytes.decode()
        # 使用 json.loads 将字符串转为 python 字典
        book_dict = json.loads(json_str)

        # 此处可以进行校验,此处详细的校验参数省略
        # 保存添加到数据库中
        # 这里可以查看传入的数据, 如果用户传入了完整信息
        # 则这里也需要保存完整信息.
        book = BookInfo.objects.create(
            btitle=book_dict.get('btitle'),
            bpub_date=book_dict.get('bpub_date')
        )

        # 返回 json 字符串和状态码
        # 梅老板提示: 创建成功的状态码是201
        return JsonResponse({
            'id': book.id,
            'btitle': book.btitle,
            'bpub_date': book.bpub_date,
            'bread': book.bread,
            'bcomment': book.bcomment,
            'image': book.image.url if book.image else ''
        }, status=201)

class BookDetailView(View):
    def get(self, request, pk):
        """
        获取单个图书信息: 用户获取某一个图书数据,所以会传入 id 值
        路由： GET  /books/<pk>/
        """

        # 首先检验凭借 pk 值, 获得的 book 是否存在
        try:
            book = BookInfo.objects.get(pk=pk)
        except BookInfo.DoesNotExist:
            # 不存在报404错误
            return HttpResponse(status=404)

        # 存在返回 json 数据
        return JsonResponse({
            'id': book.id,
            'btitle': book.btitle,
            'bpub_date': book.bpub_date,
            'bread': book.bread,
            'bcomment': book.bcomment,
            'image': book.image.url if book.image else ''
        })

    def put(self, request, pk):
        """
        修改图书信息: 修改哪一个图书信息, 所以用户传入修改的 id
        路由： PUT  /books/<pk>
        """

        # 首先检验凭借 pk 值, 获得的 book 是否存在
        try:
            book = BookInfo.objects.get(pk=pk)
        except BookInfo.DoesNotExist:
            # 不存在报404错误
            return HttpResponse(status=404)

        # 获取所有的非表单数据, 得到 bytes
        # 解码: 把 bytes 转为 str
        # 使用 json.loads 将字符串转为 python 字典
        json_bytes = request.body
        json_str = json_bytes.decode()
        book_dict = json.loads(json_str)

        # 此处可以进行校验,此处详细的校验参数省略

        # 前端获取的数据中,根据key值, 获取对应的value并赋给
        # 数据库中获取的对象, 进行覆盖操作
        # 梅老板提示: 别忘了保存
        book.btitle = book_dict.get('btitle')
        book.bpub_date = book_dict.get('bpub_date')
        book.save()

        # 存在返回 json 数据
        return JsonResponse({
            'id': book.id,
            'btitle': book.btitle,
            'bpub_date': book.bpub_date,
            'bread': book.bread,
            'bcomment': book.bcomment,
            'image': book.image.url if book.image else ''
        })


    def delete(self, request, pk):
        """
        删除图书: 删除哪一个数据, 用户需要传入 id 值
        路由： DELETE /books/<pk>/
        """

        # 首先检验凭借 pk 值, 获得的 book 是否存在
        try:
            book = BookInfo.objects.get(pk=pk)
        except BookInfo.DoesNotExist:
            # 不存在报404错误
            return HttpResponse(status=404)

        # 如果存在,调用 delete() 函数,删掉对应的内容
        book.delete()

        # 返回空文档和204
        # 梅老板提示: 
        # 204: NO CONTENT - [DELETE]：用户删除数据成功
        return HttpResponse(status=204)
